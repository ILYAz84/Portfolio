def data(df):
    features = df.drop(['id','product'], axis=1)
    target = df['product']
    features_train, features_valid, target_train, target_valid = train_test_split(
    features, target, test_size=0.25, random_state=12345) # 25%     
    return features_train, features_valid, target_train, target_valid

def rmse(df): #  ,    
    f_train, f_valid, t_train, t_valid = data(df) #   data
    model.fit(f_train,t_train) #     
    predictions_valid = model.predict(f_valid) #     
    rmse = mean_squared_error(t_valid, predictions_valid)**0.5 # RMSE   
    print('RMSE      : {:.3f}'.format(rmse))
    print('R2 =', r2_score(t_valid,predictions_valid))
    print('C   : {:.2f}(. )'.format(predictions_valid.mean()))
    print('----------')

def test_1000(df): #    1000 
    f_train, f_valid, t_train, t_valid = data(df) #   data
    model.fit(f_train,t_train) #     
    predictions_valid = model.predict(f_valid) 
    
    values = [] #  \

    for _ in range(1000): #     200  
        #  500  
        df_predict_500 = pd.Series(predictions_valid).sample(500, random_state=state)

        df_best_predict_200_index = df_predict_500.sort_values(ascending=False).head(200).index
        #  200     (   500 )
        df_best_valid_200 = t_valid.reset_index(drop=True)[df_best_predict_200_index] 
        values.append(df_best_valid_200.sum() * 450_000 - 10_000_000_000) 
    return pd.DataFrame(values)

def interval(df):
    revenue = df[0].mean()

    if  revenue > 0:
        print(' ')
        print('C  {:.0f} .'.format(revenue))
    else:
        print(' ')
        print(' {:.0f} .'.format(revenue)) 
        
    print('95%-    : ', 
          df.sort_values(by=0, ascending=False).quantile([0.025, 0.975]).values)
                   
    p1 = df[df[0] >= 0][0].count()/df.shape[0]
    print('    : {:.2f}'.format(p1))
    print(' ()    : {:.2f}'.format(1 - p1))
    print()
    print('______________________')
    print()

def interval_list(A):
    i = 0
    for x in A:
        print(': ', i)
        i += 1
        interval(x) #   interval
